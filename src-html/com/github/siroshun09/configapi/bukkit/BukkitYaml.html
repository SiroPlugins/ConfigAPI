017package com.github.siroshun09.configapi.bukkit;
018
019import com.github.siroshun09.configapi.common.util.FileUtils;
020import com.github.siroshun09.configapi.common.yaml.AbstractYaml;
021import com.github.siroshun09.configapi.common.yaml.Yaml;
022import org.bukkit.Material;
023import org.bukkit.configuration.file.YamlConfiguration;
024import org.bukkit.inventory.ItemStack;
025import org.jetbrains.annotations.NotNull;
026import org.jetbrains.annotations.Nullable;
027
028import java.io.IOException;
029import java.nio.file.Files;
030import java.nio.file.Path;
031import java.util.List;
032import java.util.Objects;
033import java.util.Set;
034
035/**
036 * This class that implements {@link Yaml} on Bukkit.
037 */
038public class BukkitYaml extends AbstractYaml {
039
040    private YamlConfiguration config = new YamlConfiguration();
041    private boolean isLoaded = false;
042
043    /**
044     * Creates a {@link BukkitYaml}.
045     *
046     * @param filePath the file to load or save.
047     */
048    public BukkitYaml(@NotNull Path filePath) {
049        super(filePath);
050    }
051
052    /**
053     * {@inheritDoc}
054     */
055    @Override
056    public void load() throws IOException {
057        FileUtils.createFileIfNotExists(filePath);
058
059        if (Files.isRegularFile(filePath) && Files.isReadable(filePath)) {
060            config = YamlConfiguration.loadConfiguration(filePath.toFile());
061            isLoaded = true;
062        }
063    }
064
065    /**
066     * {@inheritDoc}
067     */
068    @Override
069    public boolean isLoaded() {
070        return isLoaded;
071    }
072
073    /**
074     * {@inheritDoc}
075     */
076    @Override
077    public void save() throws IOException {
078        FileUtils.createFileIfNotExists(filePath);
079
080        if (Files.isRegularFile(filePath) && Files.isWritable(filePath)) {
081            config.save(filePath.toFile());
082        }
083    }
084
085    /**
086     * {@inheritDoc}
087     */
088    @Override
089    @NotNull
090    public Path getPath() {
091        return filePath;
092    }
093
094    /**
095     * Gets {@link YamlConfiguration}.
096     *
097     * @return {@link YamlConfiguration}
098     */
099    @NotNull
100    public YamlConfiguration getConfig() {
101        return config;
102    }
103
104    /**
105     * {@inheritDoc}
106     */
107    @Override
108    public @Nullable Object get(@NotNull String path) {
109        return config.get(path);
110    }
111
112    /**
113     * {@inheritDoc}
114     */
115    @Override
116    public boolean getBoolean(@NotNull String path, boolean def) {
117        Objects.requireNonNull(path, "path must not be null.");
118        return config.getBoolean(path, def);
119    }
120
121    /**
122     * {@inheritDoc}
123     */
124    @Override
125    public double getDouble(@NotNull String path, double def) {
126        Objects.requireNonNull(path, "path must not be null.");
127        return config.getDouble(path, def);
128    }
129
130    /**
131     * {@inheritDoc}
132     */
133    @Override
134    public int getInteger(@NotNull String path, int def) {
135        Objects.requireNonNull(path, "path must not be null.");
136        return config.getInt(path, def);
137    }
138
139    /**
140     * {@inheritDoc}
141     */
142    @Override
143    public long getLong(@NotNull String path, long def) {
144        Objects.requireNonNull(path, "path must not be null.");
145        return config.getLong(path, def);
146    }
147
148    /**
149     * {@inheritDoc}
150     */
151    @Override
152    @NotNull
153    public String getString(@NotNull String path, @NotNull String def) {
154        Objects.requireNonNull(path, "path must not be null.");
155        Objects.requireNonNull(def, "def must not be null.");
156
157        String value = config.getString(path, def);
158        return value != null ? value : def;
159    }
160
161    /**
162     * {@inheritDoc}
163     */
164    @Override
165    @NotNull
166    public List<String> getStringList(@NotNull String path, @NotNull List<String> def) {
167        Objects.requireNonNull(path, "path must not be null.");
168        Objects.requireNonNull(def, "def must not be null.");
169        return config.getStringList(path);
170    }
171
172    /**
173     * {@inheritDoc}
174     */
175    @Override
176    @NotNull
177    public List<Short> getShortList(@NotNull String path, @NotNull List<Short> def) {
178        Objects.requireNonNull(path, "path must not be null.");
179        Objects.requireNonNull(def, "def must not be null.");
180        return config.getShortList(path);
181    }
182
183
184    /**
185     * {@inheritDoc}
186     */
187    @Override
188    @NotNull
189    public List<Integer> getIntegerList(@NotNull String path, @NotNull List<Integer> def) {
190        Objects.requireNonNull(path, "path must not be null.");
191        Objects.requireNonNull(def, "def must not be null.");
192        return config.getIntegerList(path);
193    }
194
195
196    /**
197     * {@inheritDoc}
198     */
199    @Override
200    @NotNull
201    public List<Long> getLongList(@NotNull String path, @NotNull List<Long> def) {
202        Objects.requireNonNull(path, "path must not be null.");
203        Objects.requireNonNull(def, "def must not be null.");
204        return config.getLongList(path);
205    }
206
207
208    /**
209     * {@inheritDoc}
210     */
211    @Override
212    @NotNull
213    public List<Float> getFloatList(@NotNull String path, @NotNull List<Float> def) {
214        Objects.requireNonNull(path, "path must not be null.");
215        Objects.requireNonNull(def, "def must not be null.");
216        return config.getFloatList(path);
217    }
218
219    /**
220     * {@inheritDoc}
221     */
222    @Override
223    @NotNull
224    public List<Double> getDoubleList(@NotNull String path, @NotNull List<Double> def) {
225        Objects.requireNonNull(path, "path must not be null.");
226        Objects.requireNonNull(def, "def must not be null.");
227        return config.getDoubleList(path);
228    }
229
230    /**
231     * Gets the requested {@link ItemStack} by path.
232     * <p>
233     * If the {@link ItemStack} could not be obtained, this method returns {@code new ItemStack(Material.AIR)}.
234     *
235     * @param path Path of the {@link ItemStack} to get.
236     * @return Requested {@link ItemStack}.
237     * @since 1.5
238     */
239    @NotNull
240    public ItemStack getItemStack(@NotNull String path) {
241        return getItemStack(path, new ItemStack(Material.AIR));
242    }
243
244    /**
245     * {@inheritDoc}
246     */
247    @Override
248    @NotNull
249    public Set<String> getKeys() {
250        return config.getKeys(false);
251    }
252
253    /**
254     * {@inheritDoc}
255     */
256    @Override
257    public void set(@NotNull String path, @Nullable Object value) {
258        Objects.requireNonNull(path, "path must not be null.");
259        config.set(path, value);
260    }
261
262    /**
263     * Gets the requested {@link ItemStack} by path.
264     *
265     * @param path Path of the {@link ItemStack} to get.
266     * @param def  The default {@link ItemStack} to return if the {@link ItemStack} could not be obtained.
267     * @return Requested {@link ItemStack}.
268     * @since 1.5
269     */
270    @NotNull
271    public ItemStack getItemStack(@NotNull String path, @NotNull ItemStack def) {
272        Objects.requireNonNull(path, "path must not be null.");
273        Objects.requireNonNull(def, "def must not be null.");
274
275        ItemStack item = config.getItemStack(path);
276        return item != null ? item : def;
277    }
278}
