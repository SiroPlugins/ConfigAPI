017package com.github.siroshun09.configapi.api;
018
019import com.github.siroshun09.configapi.api.serializer.Serializer;
020import org.jetbrains.annotations.NotNull;
021import org.jetbrains.annotations.Nullable;
022import org.jetbrains.annotations.Unmodifiable;
023
024import java.util.Collections;
025import java.util.List;
026import java.util.Set;
027
028/**
029 * An interface that can get a value from a path.
030 */
031public interface Configuration {
032
033    /**
034     * The separator to split paths.
035     */
036    char PATH_SEPARATOR = '.';
037
038    /**
039     * The string representation of {@link Configuration#PATH_SEPARATOR}.
040     */
041    String PATH_SEPARATOR_STRING = String.valueOf(PATH_SEPARATOR);
042
043    /**
044     * Gets the object of the specified path.
045     * <p>
046     * If the object could not be obtained, this method returns {@code null}.
047     *
048     * @param path the path to get the object
049     * @return the object or {@code null}
050     */
051    @Nullable Object get(@NotNull String path);
052
053    /**
054     * Gets the object of the specified path.
055     * <p>
056     * If the object could not be obtained or deserialized,
057     * this method returns {@code null}.
058     *
059     * @param path       the path to get the object
060     * @param serializer the serializer to deserialize object
061     * @param <T>        the type of value
062     * @return the deserialized object or {@code null}
063     */
064    <T> @Nullable T get(@NotNull String path, @NotNull Serializer<T, ?> serializer);
065
066    /**
067     * Gets the object of the specified path.
068     *
069     * @param path the path to get the object
070     * @param def  the default object to return if the object could not be obtained
071     * @return the object
072     */
073    @NotNull Object get(@NotNull String path, @NotNull Object def);
074
075    /**
076     * Gets the object of the specified path.
077     * <p>
078     * The object to be returned will be deserialized using {@link Serializer}.
079     *
080     * @param path       the path to get the object
081     * @param serializer the serializer to deserialize the object
082     * @param def        the default object to return if the object could not be obtained or deserialized
083     * @param <T>        the type of value
084     * @return the deserialized object or default object
085     */
086    <T> @NotNull T get(@NotNull String path, @NotNull Serializer<T, ?> serializer, @NotNull T def);
087
088    /**
089     * Sets the object to the specified path.
090     * <p>
091     * If given object is {@code null},
092     * the current object of the path will be removed.
093     *
094     * @param path  the path to set or remove
095     * @param value the object to set or {@code null}
096     */
097    void set(@NotNull String path, @Nullable Object value);
098
099    /**
100     * Sets the object to the specified path.
101     * <p>
102     * The object to set will be serialized using {@link Serializer}
103     *
104     * @param path       the path to set
105     * @param value      the value to set
106     * @param serializer the serializer to serialize value
107     * @param <T>        the type of value
108     */
109    <T> void set(@NotNull String path, @NotNull T value, @NotNull Serializer<T, ?> serializer);
110
111    /**
112     * Gets the set of root paths included in this {@link Configuration}.
113     *
114     * @return set of root paths
115     */
116    @NotNull @Unmodifiable Set<String> getPaths();
117
118    /**
119     * Gets the set of root objects included in this {@link Configuration}.
120     *
121     * @return set of root keys
122     */
123    @NotNull @Unmodifiable Set<Object> getValues();
124
125    /**
126     * Gets the {@link Configuration} section of the specified path.
127     * <p>
128     * If the {@link Configuration} section could not be obtained,
129     * this method returns {@code null}.
130     *
131     * @param path the path to get the {@link Configuration} section
132     * @return the {@link Configuration} section or {@code null}
133     */
134    @Nullable Configuration getSection(@NotNull String path);
135
136    /**
137     * Gets the list of the specified path.
138     * <p>
139     * If the list could not be obtained,
140     * this method returns {@link Collections#emptyList()}.
141     *
142     * @param path the path to get the list
143     * @return the list
144     */
145    @NotNull @Unmodifiable List<?> getList(@NotNull String path);
146
147    /**
148     * Gets the list of the specified path.
149     *
150     * @param path the path to get the list
151     * @param def  the default list to return if the list could not be obtained
152     * @return the list or default one
153     */
154    @NotNull @Unmodifiable List<?> getList(@NotNull String path, @NotNull List<?> def);
155
156    /**
157     * Gets the list of the specified path.
158     * <p>
159     * Each element in the list is converted by the {@link Serializer},
160     * and those that cannot be converted are removed from the returning list.
161     * <p>
162     * The list stored in this configuration is not changed.
163     *
164     * @param path       the path to get the list
165     * @param serializer the serializer to serialize elements
166     * @param <T>        the type of value
167     * @return the serialized list
168     */
169    @NotNull @Unmodifiable <T> List<T> getList(@NotNull String path, @NotNull Serializer<T, ?> serializer);
170
171    /**
172     * Gets the list of the specified path.
173     * <p>
174     * Each element in the list is converted by the {@link Serializer},
175     * and those that cannot be converted are removed from the list to be returned.
176     * <p>
177     * The list stored in this configuration will not be changed.
178     *
179     * @param path       the path to get the list
180     * @param serializer the serializer to serialize elements
181     * @param def        the default list to return if the list could not be obtained
182     * @param <T>        the type of value
183     * @return the serialized list or default one
184     */
185    @NotNull @Unmodifiable <T> List<T> getList(@NotNull String path,
186                                               @NotNull Serializer<T, ?> serializer, List<T> def);
187
188    /**
189     * Sets the list to specified path.
190     * <p>
191     * Each element in the list is converted by the {@link Serializer},
192     * and those that cannot be converted are removed from the list to be set.
193     * <p>
194     * The list given as an argument will not be changed.
195     *
196     * @param path       the path to set the list
197     * @param list       the list to set
198     * @param serializer the serializer to serialize elements
199     * @param <T>        the type of value
200     */
201    <T> void setList(@NotNull String path, @NotNull List<T> list, @NotNull Serializer<T, ?> serializer);
202
203    /**
204     * Gets the boolean of the specified path.
205     * <p>
206     * If the boolean could not be obtained, this method returns {@code false}.
207     *
208     * @param path the path to get the boolean
209     * @return the boolean
210     */
211    boolean getBoolean(@NotNull String path);
212
213    /**
214     * Gets the boolean of the specified path.
215     *
216     * @param path the path to get the boolean
217     * @param def  the default boolean to return if the boolean could not be obtained
218     * @return the boolean
219     */
220    boolean getBoolean(@NotNull String path, boolean def);
221
222    /**
223     * Gets the byte of the specified path.
224     * <p>
225     * If the byte could not be obtained, this method returns 0.
226     *
227     * @param path the path to get the byte
228     * @return the byte
229     */
230    byte getByte(@NotNull String path);
231
232    /**
233     * Gets the byte of the specified path.
234     *
235     * @param path the path to get the byte
236     * @param def  the default byte to return if the byte could not be obtained
237     * @return the byte
238     */
239    byte getByte(@NotNull String path, byte def);
240
241    /**
242     * Gets the double of the specified path.
243     * <p>
244     * If the double could not be obtained, this method returns 0.
245     *
246     * @param path the path to get the double
247     * @return the double
248     */
249    double getDouble(@NotNull String path);
250
251    /**
252     * Gets the double of the specified path.
253     *
254     * @param path the path to get the double
255     * @param def  the default double to return if the double could not be obtained
256     * @return the double
257     */
258    double getDouble(@NotNull String path, double def);
259
260    /**
261     * Gets the float of the specified path.
262     * <p>
263     * If the float could not be obtained, this method returns 0.
264     *
265     * @param path the path to get the float
266     * @return the float
267     */
268    float getFloat(@NotNull String path);
269
270    /**
271     * Gets the float of the specified path.
272     *
273     * @param path the path to get the float
274     * @param def  the default float to return if the float could not be obtained
275     * @return the float
276     */
277    float getFloat(@NotNull String path, float def);
278
279    /**
280     * Gets the integer of the specified path.
281     * <p>
282     * If the integer could not be obtained, this method returns 0.
283     *
284     * @param path the path to get the integer
285     * @return the integer
286     */
287    int getInteger(@NotNull String path);
288
289    /**
290     * Gets the integer of the specified path.
291     *
292     * @param path the path to get the integer
293     * @param def  the default integer to return if the integer could not be obtained
294     * @return the integer
295     */
296    int getInteger(@NotNull String path, int def);
297
298    /**
299     * Gets the long of the specified path.
300     * <p>
301     * If the long could not be obtained, this method returns 0.
302     *
303     * @param path the path to get the long
304     * @return the long
305     */
306    long getLong(@NotNull String path);
307
308    /**
309     * Gets the long of the specified path.
310     *
311     * @param path the path to get the long
312     * @param def  the default long to return if the long could not be obtained
313     * @return the long
314     */
315    long getLong(@NotNull String path, long def);
316
317    /**
318     * Gets the short of the specified path.
319     * <p>
320     * If the short could not be obtained, this method returns 0.
321     *
322     * @param path the path to get the short
323     * @return the short
324     */
325    short getShort(@NotNull String path);
326
327    /**
328     * Gets the short of the specified path.
329     *
330     * @param path the path to get the short
331     * @param def  the default short to return if the short could not be obtained
332     * @return the short
333     */
334    short getShort(@NotNull String path, short def);
335
336    /**
337     * Gets the string of the specified path.
338     * <p>
339     * If the string could not be obtained, this method returns 0.
340     *
341     * @param path the path to get the string
342     * @return the string
343     */
344    @NotNull String getString(@NotNull String path);
345
346    /**
347     * Gets the string of the specified path.
348     *
349     * @param path the path to get the string
350     * @param def  the default string to return if the string could not be obtained
351     * @return the string
352     */
353    @NotNull String getString(@NotNull String path, @NotNull String def);
354
355    /**
356     * Gets the boolean list of the specified path.
357     * <p>
358     * If the boolean list could not be obtained,
359     * this method returns {@link Collections#emptyList()}.
360     *
361     * @param path the path to get the boolean list
362     * @return the boolean list
363     */
364    @NotNull @Unmodifiable List<Boolean> getBooleanList(@NotNull String path);
365
366    /**
367     * Gets the boolean list of the specified path.
368     *
369     * @param path the path to get the boolean list
370     * @param def  the default boolean list to return if the boolean list could not be obtained
371     * @return the boolean list
372     */
373    @NotNull @Unmodifiable List<Boolean> getBooleanList(@NotNull String path, @NotNull List<Boolean> def);
374
375    /**
376     * Gets the byte list of the specified path.
377     * <p>
378     * If the byte list could not be obtained,
379     * this method returns {@link Collections#emptyList()}.
380     *
381     * @param path the path to get the byte list
382     * @return the byte list
383     */
384    @NotNull @Unmodifiable List<Byte> getByteList(@NotNull String path);
385
386    /**
387     * Gets the byte list of the specified path.
388     *
389     * @param path the path to get the byte list
390     * @param def  the default byte list to return if the byte list could not be obtained
391     * @return the byte list
392     */
393    @NotNull @Unmodifiable List<Byte> getByteList(@NotNull String path, @NotNull List<Byte> def);
394
395    /**
396     * Gets the double list of the specified path.
397     * <p>
398     * If the double list could not be obtained,
399     * this method returns {@link Collections#emptyList()}.
400     *
401     * @param path the path to get the double list
402     * @return the double list
403     */
404    @NotNull @Unmodifiable List<Double> getDoubleList(@NotNull String path);
405
406    /**
407     * Gets the double list of the specified path.
408     *
409     * @param path the path to get the double list
410     * @param def  the default double list to return if the double list could not be obtained
411     * @return the double list
412     */
413    @NotNull @Unmodifiable List<Double> getDoubleList(@NotNull String path, @NotNull List<Double> def);
414
415    /**
416     * Gets the float list of the specified path.
417     * <p>
418     * If the float list could not be obtained,
419     * this method returns {@link Collections#emptyList()}.
420     *
421     * @param path the path to get the float list
422     * @return the float list
423     */
424    @NotNull @Unmodifiable List<Float> getFloatList(@NotNull String path);
425
426    /**
427     * Gets the float list of the specified path.
428     *
429     * @param path the path to get the float list
430     * @param def  the default float list to return if the float list could not be obtained
431     * @return the float list
432     */
433    @NotNull @Unmodifiable List<Float> getFloatList(@NotNull String path, @NotNull List<Float> def);
434
435    /**
436     * Gets the integer list of the specified path.
437     * <p>
438     * If the integer list could not be obtained,
439     * this method returns {@link Collections#emptyList()}.
440     *
441     * @param path the path to get the integer list
442     * @return the integer list
443     */
444    @NotNull @Unmodifiable List<Integer> getIntegerList(@NotNull String path);
445
446    /**
447     * Gets the integer list of the specified path.
448     *
449     * @param path the path to get the integer list
450     * @param def  the default integer list to return if the integer list could not be obtained
451     * @return the integer list
452     */
453    @NotNull @Unmodifiable List<Integer> getIntegerList(@NotNull String path, @NotNull List<Integer> def);
454
455    /**
456     * Gets the long list of the specified path.
457     * <p>
458     * If the long list could not be obtained,
459     * this method returns {@link Collections#emptyList()}.
460     *
461     * @param path the path to get the long list
462     * @return the long list
463     */
464    @NotNull @Unmodifiable List<Long> getLongList(@NotNull String path);
465
466    /**
467     * Gets the long list of the specified path.
468     *
469     * @param path the path to get the long list
470     * @param def  the default long list to return if the long list could not be obtained
471     * @return the long list
472     */
473    @NotNull @Unmodifiable List<Long> getLongList(@NotNull String path, @NotNull List<Long> def);
474
475    /**
476     * Gets the short list of the specified path.
477     * <p>
478     * If the short list could not be obtained,
479     * this method returns {@link Collections#emptyList()}.
480     *
481     * @param path the path to get the short list
482     * @return the short list
483     */
484    @NotNull @Unmodifiable List<Short> getShortList(@NotNull String path);
485
486    /**
487     * Gets the short list of the specified path.
488     *
489     * @param path the path to get the short list
490     * @param def  the default short list to return if the short list could not be obtained
491     * @return the short list
492     */
493    @NotNull @Unmodifiable List<Short> getShortList(@NotNull String path, @NotNull List<Short> def);
494
495    /**
496     * Gets the string list of the specified path.
497     * <p>
498     * If the string list could not be obtained,
499     * this method returns {@link Collections#emptyList()}.
500     *
501     * @param path the path to get the string list
502     * @return the string list
503     */
504    @NotNull @Unmodifiable List<String> getStringList(@NotNull String path);
505
506    /**
507     * Gets the string list of the specified path.
508     *
509     * @param path the path to get the string list
510     * @param def  the default string list to return if the string list could not be obtained
511     * @return the string list
512     */
513    @NotNull @Unmodifiable List<String> getStringList(@NotNull String path, @NotNull List<String> def);
