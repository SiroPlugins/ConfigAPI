017package com.github.siroshun09.configapi.api;
018
019import com.github.siroshun09.configapi.api.serializer.Serializer;
020import org.jetbrains.annotations.NotNull;
021import org.jetbrains.annotations.Nullable;
022import org.jetbrains.annotations.Unmodifiable;
023
024import java.util.Collections;
025import java.util.List;
026import java.util.Objects;
027import java.util.function.Function;
028import java.util.stream.Collectors;
029import java.util.stream.Stream;
030
031/**
032 * An abstract implementation of {@link Configuration}.
033 */
034public abstract class AbstractConfiguration implements Configuration {
035
036    @Override
037    public @NotNull Object get(@NotNull String path, @NotNull Object def) {
038        var value = get(path);
039        return value != null ? value : Objects.requireNonNull(path);
040    }
041
042    @Override
043    public <T> @Nullable T get(@NotNull String path, @NotNull Serializer<T, ?> serializer) {
044        var value = get(path);
045        return value != null ? serializer.deserialize(value) : null;
046    }
047
048    @Override
049    public <T> @NotNull T get(@NotNull String path, @NotNull Serializer<T, ?> serializer, @NotNull T def) {
050        var value = get(path, serializer);
051        return value != null ? value : def;
052    }
053
054    @Override
055    public <T> void set(@NotNull String path, @NotNull T value, @NotNull Serializer<T, ?> serializer) {
056        set(path, serializer.serialize(value));
057    }
058
059    @Override
060    public @NotNull @Unmodifiable List<?> getList(@NotNull String path) {
061        return getList(path, Collections.emptyList());
062    }
063
064    @Override
065    public @NotNull @Unmodifiable List<?> getList(@NotNull String path, @NotNull List<?> def) {
066        var value = get(path);
067        return value instanceof List<?> ? (List<?>) value : def;
068    }
069
070    @Override
071    public @NotNull @Unmodifiable <T> List<T> getList(@NotNull String path, @NotNull Serializer<T, ?> serializer) {
072        return getList(path).stream()
073                .map(serializer::deserialize)
074                .filter(Objects::nonNull)
075                .collect(Collectors.toUnmodifiableList());
076    }
077
078    @Override
079    public @NotNull @Unmodifiable <T> List<T> getList(@NotNull String path, @NotNull Serializer<T, ?> serializer, List<T> def) {
080        var list = getListOrNull(path);
081
082        if (list != null) {
083            return list.stream()
084                    .map(serializer::deserialize)
085                    .filter(Objects::nonNull)
086                    .collect(Collectors.toUnmodifiableList());
087        } else {
088            return def;
089        }
090    }
091
092    @Override
093    public <T> void setList(@NotNull String path, @NotNull List<T> list, @NotNull Serializer<T, ?> serializer) {
094        set(path, list.stream().map(serializer::serialize).collect(Collectors.toUnmodifiableList()));
095    }
096
097    @Override
098    public boolean getBoolean(@NotNull String path) {
099        return getBoolean(path, false);
100    }
101
102    @Override
103    public boolean getBoolean(@NotNull String path, boolean def) {
104        var value = get(path);
105        return value instanceof Boolean ? (Boolean) value : def;
106    }
107
108    @Override
109    public byte getByte(@NotNull String path) {
110        return getByte(path, (byte) 0);
111    }
112
113    @Override
114    public byte getByte(@NotNull String path, byte def) {
115        return getNumber(path, def, Number::byteValue);
116    }
117
118    @Override
119    public double getDouble(@NotNull String path) {
120        return getDouble(path, 0);
121    }
122
123    @Override
124    public double getDouble(@NotNull String path, double def) {
125        return getNumber(path, def, Number::doubleValue);
126    }
127
128    @Override
129    public float getFloat(@NotNull String path) {
130        return getFloat(path, 0);
131    }
132
133    @Override
134    public float getFloat(@NotNull String path, float def) {
135        return getNumber(path, def, Number::floatValue);
136    }
137
138    @Override
139    public int getInteger(@NotNull String path) {
140        return getInteger(path, 0);
141    }
142
143    @Override
144    public int getInteger(@NotNull String path, int def) {
145        return getNumber(path, def, Number::intValue);
146    }
147
148    @Override
149    public long getLong(@NotNull String path) {
150        return getLong(path, 0);
151    }
152
153    @Override
154    public long getLong(@NotNull String path, long def) {
155        return getNumber(path, def, Number::longValue);
156    }
157
158    @Override
159    public short getShort(@NotNull String path) {
160        return getShort(path, (short) 0);
161    }
162
163    @Override
164    public short getShort(@NotNull String path, short def) {
165        return getNumber(path, def, Number::shortValue);
166    }
167
168    @Override
169    public @NotNull String getString(@NotNull String path) {
170        return getString(path, "");
171    }
172
173    @Override
174    public @NotNull String getString(@NotNull String path, @NotNull String def) {
175        var value = get(path);
176        return value instanceof String ? (String) value : def;
177    }
178
179    @Override
180    public @NotNull @Unmodifiable List<Boolean> getBooleanList(@NotNull String path) {
181        return getBooleanList(path, Collections.emptyList());
182    }
183
184    @Override
185    public @NotNull @Unmodifiable List<Boolean> getBooleanList(@NotNull String path, @NotNull List<Boolean> def) {
186        var list = getListOrNull(path);
187
188        if (list != null) {
189            return list.stream()
190                    .filter(object -> object instanceof Boolean)
191                    .map(object -> (Boolean) object)
192                    .collect(Collectors.toUnmodifiableList());
193        } else {
194            return def;
195        }
196    }
197
198    @Override
199    public @NotNull @Unmodifiable List<Byte> getByteList(@NotNull String path) {
200        return getByteList(path, Collections.emptyList());
201    }
202
203    @Override
204    public @NotNull @Unmodifiable List<Byte> getByteList(@NotNull String path, @NotNull List<Byte> def) {
205        var stream = getNumberStreamOrNull(path);
206
207        if (stream != null) {
208            return stream.map(Number::byteValue).collect(Collectors.toUnmodifiableList());
209        } else {
210            return def;
211        }
212    }
213
214    @Override
215    public @NotNull @Unmodifiable List<Double> getDoubleList(@NotNull String path) {
216        return getDoubleList(path, Collections.emptyList());
217    }
218
219    @Override
220    public @NotNull @Unmodifiable List<Double> getDoubleList(@NotNull String path, @NotNull List<Double> def) {
221        var stream = getNumberStreamOrNull(path);
222
223        if (stream != null) {
224            return stream.map(Number::doubleValue).collect(Collectors.toUnmodifiableList());
225        } else {
226            return def;
227        }
228    }
229
230    @Override
231    public @NotNull @Unmodifiable List<Float> getFloatList(@NotNull String path) {
232        return getFloatList(path, Collections.emptyList());
233    }
234
235    @Override
236    public @NotNull @Unmodifiable List<Float> getFloatList(@NotNull String path, @NotNull List<Float> def) {
237        var stream = getNumberStreamOrNull(path);
238
239        if (stream != null) {
240            return stream.map(Number::floatValue).collect(Collectors.toUnmodifiableList());
241        } else {
242            return def;
243        }
244    }
245
246    @Override
247    public @NotNull @Unmodifiable List<Integer> getIntegerList(@NotNull String path) {
248        return getIntegerList(path, Collections.emptyList());
249    }
250
251    @Override
252    public @NotNull @Unmodifiable List<Integer> getIntegerList(@NotNull String path, @NotNull List<Integer> def) {
253        var stream = getNumberStreamOrNull(path);
254
255        if (stream != null) {
256            return stream.map(Number::intValue).collect(Collectors.toUnmodifiableList());
257        } else {
258            return def;
259        }
260    }
261
262    @Override
263    public @NotNull @Unmodifiable List<Long> getLongList(@NotNull String path) {
264        return getLongList(path, Collections.emptyList());
265    }
266
267    @Override
268    public @NotNull @Unmodifiable List<Long> getLongList(@NotNull String path, @NotNull List<Long> def) {
269        var stream = getNumberStreamOrNull(path);
270
271        if (stream != null) {
272            return stream.map(Number::longValue).collect(Collectors.toUnmodifiableList());
273        } else {
274            return def;
275        }
276    }
277
278    @Override
279    public @NotNull @Unmodifiable List<Short> getShortList(@NotNull String path) {
280        return getShortList(path, Collections.emptyList());
281    }
282
283    @Override
284    public @NotNull @Unmodifiable List<Short> getShortList(@NotNull String path, @NotNull List<Short> def) {
285        var stream = getNumberStreamOrNull(path);
286
287        if (stream != null) {
288            return stream.map(Number::shortValue).collect(Collectors.toUnmodifiableList());
289        } else {
290            return def;
291        }
292    }
293
294    @Override
295    public @NotNull @Unmodifiable List<String> getStringList(@NotNull String path) {
296        return getStringList(path, Collections.emptyList());
297    }
298
299    @Override
300    public @NotNull @Unmodifiable List<String> getStringList(@NotNull String path, @NotNull List<String> def) {
301        var list = getListOrNull(path);
302
303        if (list != null) {
304            return list.stream()
305                    .map(object -> object instanceof String ? (String) object : object.toString())
306                    .collect(Collectors.toUnmodifiableList());
307        } else {
308            return def;
309        }
310    }
311
312    private <T extends Number> @NotNull T getNumber(@NotNull String path, T def,
313                                                    @NotNull Function<Number, T> converter) {
314        var value = get(path);
315        return value instanceof Number ? converter.apply((Number) value) : def;
316    }
317
318    private @Nullable List<?> getListOrNull(@NotNull String path) {
319        var value = get(path);
320        return value instanceof List<?> ? (List<?>) value : null;
321    }
322
323    private @Nullable Stream<Number> getNumberStreamOrNull(@NotNull String path) {
324        var list = getListOrNull(path);
325
326        if (list != null) {
327            return list.stream()
328                    .filter(object -> object instanceof Number)
329                    .map(object -> (Number) object);
330        } else {
331            return null;
332        }
333    }
