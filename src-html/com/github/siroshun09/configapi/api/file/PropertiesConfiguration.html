017package com.github.siroshun09.configapi.api.file;
018
019import com.github.siroshun09.configapi.api.Configuration;
020import org.jetbrains.annotations.NotNull;
021import org.jetbrains.annotations.Nullable;
022import org.jetbrains.annotations.Unmodifiable;
023
024import java.io.IOException;
025import java.nio.file.Files;
026import java.nio.file.Path;
027import java.util.Objects;
028import java.util.Properties;
029import java.util.Set;
030import java.util.stream.Collectors;
031
032/**
033 * A implementation class that gets value from {@link Properties}.
034 */
035public class PropertiesConfiguration extends AbstractFileConfiguration {
036
037    /**
038     * Creates a new {@link PropertiesConfiguration}.
039     *
040     * @param path the filepath to load
041     * @return the new {@link PropertiesConfiguration}
042     */
043    public static @NotNull PropertiesConfiguration create(@NotNull Path path) {
044        return create(path, new Properties());
045    }
046
047    /**
048     * Creates a new {@link PropertiesConfiguration}.
049     *
050     * @param path       the filepath to load
051     * @param properties the {@link Properties}
052     * @return the new {@link PropertiesConfiguration}
053     */
054    public static @NotNull PropertiesConfiguration create(@NotNull Path path, @NotNull Properties properties) {
055        return new PropertiesConfiguration(path, properties);
056    }
057
058    private final Properties properties;
059
060    private PropertiesConfiguration(@NotNull Path path, @NotNull Properties properties) {
061        super(path);
062        this.properties = properties;
063    }
064
065    @Override
066    public @Nullable Object get(@NotNull String path) {
067        return properties.get(path);
068    }
069
070    @Override
071    public void set(@NotNull String path, @Nullable Object value) {
072        boolean remove = value == null;
073
074        if (remove) {
075            properties.remove(path);
076        } else {
077            properties.setProperty(path, value instanceof String ? (String) value : value.toString());
078        }
079    }
080
081    @Override
082    public @NotNull @Unmodifiable Set<String> getPaths() {
083        return properties.keySet()
084                .stream()
085                .map(object -> object instanceof String ? (String) object : object.toString())
086                .collect(Collectors.toUnmodifiableSet());
087    }
088
089    @Override
090    public @NotNull @Unmodifiable Set<Object> getValues() {
091        return properties.values().stream().collect(Collectors.toUnmodifiableSet());
092    }
093
094    /**
095     * @param path the path to get the {@link Configuration} section
096     * @return {@code null}
097     * @deprecated This method always returns {@code null}.
098     */
099    @Override
100    @Deprecated
101    public @Nullable Configuration getSection(@NotNull String path) {
102        return null;
103    }
104
105    @Override
106    public void load() throws IOException {
107        properties.clear();
108
109        if (!Files.isRegularFile(getPath())) {
110            return;
111        }
112
113        try (var reader = Files.newBufferedReader(getPath())) {
114            properties.load(reader);
115        }
116
117        setLoaded(true);
118    }
119
120    @Override
121    public void save() throws IOException {
122        save(null);
123    }
124
125    /**
126     * Saves {@link Properties} with comments.
127     *
128     * @param comments the comments or {@code null}
129     * @throws IOException if an I/O error occurs
130     */
131    public void save(@Nullable String comments) throws IOException {
132        try (var writer = Files.newBufferedWriter(getPath())) {
133            properties.store(writer, comments);
134        }
135    }
136
137    @Override
138    public boolean equals(Object o) {
139        if (this == o) return true;
140        if (o == null || getClass() != o.getClass()) return false;
141        PropertiesConfiguration that = (PropertiesConfiguration) o;
142        return properties.equals(that.properties);
143    }
144
145    @Override
146    public int hashCode() {
147        return Objects.hash(properties);
148    }
149
150    @Override
151    public String toString() {
152        return "PropertiesFileConfiguration{" +
153                "properties=" + properties +
154                '}';
155    }
156}
