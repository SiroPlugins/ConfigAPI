017package com.github.siroshun09.configapi.api.util;
018
019import org.jetbrains.annotations.NotNull;
020
021import java.io.IOException;
022import java.io.InputStream;
023import java.nio.file.Files;
024import java.nio.file.Path;
025import java.util.Objects;
026import java.util.jar.JarFile;
027
028/**
029 * A utility class that provides methods to copy files contained in a jar.
030 */
031public final class ResourceUtils {
032
033    /**
034     * Copies a file from classloader.
035     *
036     * @param loader the classloader
037     * @param name   the filename
038     * @param target the filepath to save
039     * @throws IOException          if an I/O error occurs
040     * @throws NullPointerException if {@code null} is specified as an argument.
041     */
042    public static void copyFromClassLoader(@NotNull ClassLoader loader,
043                                           @NotNull String name, @NotNull Path target) throws IOException {
044        Objects.requireNonNull(loader);
045        Objects.requireNonNull(name);
046        Objects.requireNonNull(target);
047
048        try (InputStream input = loader.getResourceAsStream(name)) {
049            if (input != null) {
050                Files.copy(input, target);
051            }
052        }
053    }
054
055    /**
056     * Copies a file from classloader if the {@code target} not exists.
057     *
058     * @param loader the classloader
059     * @param name   the filename
060     * @param target the filepath to save
061     * @throws IOException          if an I/O error occurs
062     * @throws NullPointerException if {@code null} is specified as an argument.
063     */
064    public static void copyFromClassLoaderIfNotExists(@NotNull ClassLoader loader,
065                                                      @NotNull String name, @NotNull Path target) throws IOException {
066        Objects.requireNonNull(target);
067
068        if (!Files.exists(target)) {
069            copyFromClassLoader(loader, name, target);
070        }
071    }
072
073    /**
074     * Copies a file from jar.
075     *
076     * @param jar    the jar file
077     * @param name   the filename
078     * @param target the filepath to save
079     * @throws IOException          if an I/O error occurs
080     * @throws NullPointerException if {@code null} is specified as an argument.
081     */
082    public static void copyFromJar(@NotNull JarFile jar,
083                                   @NotNull String name, @NotNull Path target) throws IOException {
084        Objects.requireNonNull(jar);
085        Objects.requireNonNull(name);
086        Objects.requireNonNull(target);
087
088        var file = jar.getEntry(name);
089
090        if (file == null) {
091            return;
092        }
093
094        try (var input = jar.getInputStream(file)) {
095            Files.copy(input, target);
096        }
097    }
098
099    /**
100     * Copies a file from jar if the {@code target} not exists.
101     *
102     * @param jar    the jar file
103     * @param name   the filename
104     * @param target the filepath to save
105     * @throws IOException          if an I/O error occurs
106     * @throws NullPointerException if {@code null} is specified as an argument.
107     */
108    public static void copyFromJarIfNotExists(@NotNull JarFile jar,
109                                              @NotNull String name, @NotNull Path target) throws IOException {
110        Objects.requireNonNull(target);
111
112        if (!Files.exists(target)) {
113            copyFromJar(jar, name, target);
114        }
115    }
116
117    /**
118     * Copies a file from jar.
119     *
120     * @param jarPath the jar filepath
121     * @param name    the filename
122     * @param target  the filepath to save
123     * @throws IOException          if an I/O error occurs
124     * @throws NullPointerException if {@code null} is specified as an argument.
125     */
126    public static void copyFromJar(@NotNull Path jarPath,
127                                   @NotNull String name, @NotNull Path target) throws IOException {
128        Objects.requireNonNull(jarPath);
129
130        if (!Files.exists(jarPath)) {
131            return;
132        }
133
134        var jar = new JarFile(jarPath.toFile(), false);
135        copyFromJar(jar, name, target);
136    }
137
138    /**
139     * Copies a file from jar if the {@code target} not exists.
140     *
141     * @param jarPath the jar filepath
142     * @param name    the filename
143     * @param target  the filepath to save
144     * @throws IOException          if an I/O error occurs
145     * @throws NullPointerException if {@code null} is specified as an argument.
146     */
147    public static void copyFromJarIfNotExists(@NotNull Path jarPath,
148                                              @NotNull String name, @NotNull Path target) throws IOException {
149        Objects.requireNonNull(target);
150
151        if (!Files.exists(target)) {
152            copyFromJar(jarPath, name, target);
153        }
154    }
155
156    private ResourceUtils() {
157        throw new UnsupportedOperationException();
158    }
159}
