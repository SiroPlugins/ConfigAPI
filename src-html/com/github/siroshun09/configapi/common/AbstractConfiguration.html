017package com.github.siroshun09.configapi.common;
018
019import com.github.siroshun09.configapi.common.serialize.Serializer;
020import org.jetbrains.annotations.NotNull;
021import org.jetbrains.annotations.Nullable;
022
023import java.util.Collection;
024import java.util.LinkedHashMap;
025import java.util.Map;
026import java.util.Objects;
027
028public abstract class AbstractConfiguration implements Configuration {
029
030    protected static final String KEY_SEPARATOR_STRING = String.valueOf(KEY_SEPARATOR);
031    protected static final String KEY_SEPARATOR_REGEX = "\\" + KEY_SEPARATOR;
032
033    private final Map<String, Object> map;
034
035    protected AbstractConfiguration() {
036        map = new LinkedHashMap<>();
037    }
038
039    protected AbstractConfiguration(@NotNull Map<String, Object> map) {
040        this.map = Objects.requireNonNull(map);
041    }
042
043    protected AbstractConfiguration(@NotNull AbstractConfiguration other) {
044        this.map = other.getMap();
045    }
046
047    protected @NotNull Map<String, Object> getMap() {
048        return map;
049    }
050
051    @Override
052    public @Nullable Object get(@NotNull String path) {
053        checkPath(path);
054
055        if (!path.contains(KEY_SEPARATOR_STRING)) {
056            return getValue(path);
057        }
058
059        String[] keys = path.split(KEY_SEPARATOR_REGEX);
060        int lastIndex = keys.length - 1;
061
062        AbstractConfiguration parent = getParentOfPath(keys, lastIndex);
063
064        return parent != null ? parent.getValue(keys[lastIndex]) : null;
065    }
066
067    @Override
068    public @NotNull Collection<String> getKeys() {
069        return map.keySet();
070    }
071
072    @Override
073    public void set(@NotNull String path, @Nullable Object value) {
074        checkPath(path);
075
076        boolean remove = value == null;
077
078        if (!path.contains(KEY_SEPARATOR_STRING)) {
079            if (remove) {
080                removeValue(path);
081            } else {
082                putValue(path, value);
083            }
084        }
085
086        String[] keys = path.split(KEY_SEPARATOR_REGEX);
087        int lastIndex = keys.length - 1;
088
089        if (remove) {
090            AbstractConfiguration parent = getParentOfPath(keys, lastIndex);
091
092            if (parent != null) {
093                parent.removeValue(checkKey(keys[lastIndex]));
094            }
095        } else {
096            getOrCreateParentOfPath(keys, lastIndex).putValue(keys[lastIndex], value);
097        }
098    }
099
100    @Override
101    public <T> void set(@NotNull String path, @NotNull T value, @NotNull Serializer<T> serializer) {
102        checkPath(path);
103        Objects.requireNonNull(value);
104        Objects.requireNonNull(serializer);
105
106        Configuration serialized = serializer.serialize(value);
107
108        if (serialized instanceof AbstractConfiguration) {
109            set(path, serialized);
110        }
111    }
112
113    @Override
114    public @Nullable Configuration getSection(@NotNull String path) {
115        checkPath(path);
116
117        if (!path.contains(KEY_SEPARATOR_STRING)) {
118            return getDirectChild(path);
119        }
120
121        String[] keys = path.split(KEY_SEPARATOR_REGEX);
122        AbstractConfiguration current = this;
123
124        for (String key : keys) {
125            current = current.getDirectChild(key);
126
127            if (current == null) {
128                return null;
129            }
130        }
131
132        return current;
133    }
134
135    @Override
136    public String toString() {
137        return "ConfigurationImpl{" +
138                "map=" + map +
139                '}';
140    }
141
142    @Override
143    public boolean equals(Object o) {
144        if (this == o) return true;
145        if (!(o instanceof AbstractConfiguration)) return false;
146        AbstractConfiguration that = (AbstractConfiguration) o;
147        return getMap().equals(that.getMap());
148    }
149
150    @Override
151    public int hashCode() {
152        return Objects.hash(getMap());
153    }
154
155    private @Nullable Object getValue(@NotNull String key) {
156        return map.get(checkKey(key));
157    }
158
159    private void putValue(@NotNull String key, @NotNull Object object) {
160        if (object instanceof AbstractConfiguration) {
161            map.put(checkKey(key), ((AbstractConfiguration) object).map);
162        } else {
163            map.put(checkKey(key), object);
164        }
165    }
166
167    private void removeValue(@NotNull String key) {
168        map.remove(checkKey(key));
169    }
170
171    @SuppressWarnings("unchecked")
172    private @Nullable AbstractConfiguration getDirectChild(@NotNull String key) {
173        Object object = get(key);
174
175        if (object instanceof Map) {
176            return new ConfigurationImpl((Map<String, Object>) object);
177        } else {
178            return null;
179        }
180    }
181
182    private @Nullable AbstractConfiguration getParentOfPath(@NotNull String[] keys, int lastIndex) {
183        AbstractConfiguration current = this;
184
185        for (int i = 0; i < lastIndex && current != null; i++) {
186            current = current.getDirectChild(keys[i]);
187        }
188
189        return current;
190    }
191
192    private @NotNull AbstractConfiguration getOrCreateParentOfPath(@NotNull String[] keys, int lastIndex) {
193        AbstractConfiguration parent = this;
194        AbstractConfiguration current;
195
196        for (int i = 0; i < lastIndex; i++) {
197            current = parent.getDirectChild(keys[i]);
198
199            if (current == null) {
200                Map<String, Object> childMap = new LinkedHashMap<>();
201                parent.putValue(checkKey(keys[i]), childMap);
202                current = new ConfigurationImpl(childMap);
203            }
204
205            parent = current;
206        }
207
208        return parent;
209    }
210
211    private void checkPath(String path) {
212        if (path == null || path.isEmpty() ||
213                path.startsWith(KEY_SEPARATOR_STRING) || path.endsWith(KEY_SEPARATOR_STRING)) {
214            throw new IllegalArgumentException(path + " is invalid key.");
215        }
216    }
217
218    private @NotNull String checkKey(String key) {
219        if (key == null || key.isEmpty() || key.contains(KEY_SEPARATOR_STRING)) {
220            throw new IllegalArgumentException(key + " is invalid key.");
221        }
222
223        return key;
224    }
225}
